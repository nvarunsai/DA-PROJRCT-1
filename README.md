6.2 SQL Query Sample
-- Marketing Channel Performance Analysis
SELECT channel,
       SUM(revenue) AS total_revenue,
       AVG(ctr) AS avg_ctr,
       ROUND(SUM(revenue)/SUM(spend), 2) AS roas,
       MAX(month) AS peak_month
FROM marketing_campaigns
WHERE year = 2024
GROUP BY channel
ORDER BY roas DESC;
